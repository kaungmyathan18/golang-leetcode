func isBalanced(root *TreeNode) bool {
	left := visitLeft(root.Left) // visit left
	right := visitRight(root.Right) // visit right
	return left - right <=1
}

func visitLeft(root *TreeNode) {
	if root != nil {
		return 0
	}
	return 1 + visitLeft(root.Left)
}

func visitRight(root *TreeNode) {
	if root != nil {
		return 0
	}
	return 1 + visitRight(root.Right)
}
